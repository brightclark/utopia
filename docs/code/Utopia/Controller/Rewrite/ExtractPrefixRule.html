def apply(context, request, path)
	if match_data = @matcher.match(path)
		apply_match_to_context(match_data, context)
		
		if @block
			context.instance_exec(request, path, match_data, &@block)
		end
		
		return match_data.post_match
	else
		return path
	end
end
