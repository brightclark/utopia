def initialize(code, valid_range = 100...600)
	if code.is_a? Symbol
		code = STATUS_CODES[code]
	end
	
	unless valid_range.include? code
		raise ArgumentError.new("Status must be in range #{valid_range}, was given #{code}!")
	end
	
	@code = code
end
